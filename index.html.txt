                    document.body.style.overflowY = 'auto'; // Re-enable scrolling
                    // Start main animations after preloader is gone
                    setupScrollReveal();
                    runHeroTextAnimation();
                }, 1000); // 1 second matches the CSS transition duration
            }, 1500);
        }
        
        // --- 2. Hero Text Reveal Animation ---
        function runHeroTextAnimation() {
            const titleElement = document.getElementById('hero-title');
            if (!titleElement) return;

            const text = titleElement.textContent;
            titleElement.innerHTML = '';

            // Split the text into individual characters and wrap them in spans
            text.split('').forEach((char, index) => {
                const span = document.createElement('span');
                span.textContent = char === ' ' ? '\u00A0' : char; 
                span.style.animationDelay = `${0.03 * index}s`;
                titleElement.appendChild(span);
            });
        }
        
        // --- 3. Scroll Reveal Animation using Intersection Observer ---
        function setupScrollReveal() {
            const revealElements = document.querySelectorAll('.scroll-reveal');

            const observer = new IntersectionObserver((entries, observer) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        const delay = parseFloat(entry.target.getAttribute('data-scroll-delay')) || 0;
                        setTimeout(() => {
                            entry.target.classList.add('visible');
                        }, delay * 1000); 
                        observer.unobserve(entry.target);
                    }
                });
            }, {
                rootMargin: '0px',
                threshold: 0.1 
            });

            revealElements.forEach(el => observer.observe(el));
        }

        // --- 4. Custom Cursor Logic (Desktop Only) ---
        function setupCustomCursor() {
            const cursor = document.getElementById('custom-cursor');
            if (!cursor) return;

            // Follow the mouse
            document.addEventListener('mousemove', (e) => {
                cursor.style.left = `${e.clientX}px`;
                cursor.style.top = `${e.clientY}px`;
            });

            // Handle hover states for interactive elements (links, buttons, inputs)
            const interactiveElements = document.querySelectorAll('a, button, input, textarea, [data-hover-cursor]');

            interactiveElements.forEach(el => {
                el.addEventListener('mouseenter', () => cursor.classList.add('hovered-element'));
                el.addEventListener('mouseleave', () => cursor.classList.remove('hovered-element'));
            });
        }

        // --- 5. Parallax Mouse Effect for Hero ---
        function setupParallaxHero() {
            const target = document.getElementById('hero-title');
            if (!target) return;

            const strength = parseFloat(target.getAttribute('data-parallax-strength')) || 0.05;

            document.addEventListener('mousemove', (e) => {
                // Calculate position relative to the center of the viewport
                const x = (e.clientX - window.innerWidth / 2) * strength;
                const y = (e.clientY - window.innerHeight / 2) * strength;

                // Apply slight perspective and rotation for a 3D feel
                target.style.transform = `
                    translate3d(${-x}px, ${-y}px, 0)
                    rotateX(${-y * 0.1}deg) 
                    rotateY(${x * 0.1}deg)
                `;
                target.style.transition = 'transform 0.1s ease-out';
            });
        }

        // --- 6. 3D Tilt Effect for Process Cards ---
        function setupTiltEffect() {
            const cards = document.querySelectorAll('.tilt-card');
            
            cards.forEach(card => {
                const maxTilt = parseFloat(card.getAttribute('data-tilt-max')) || 10;
                
                // Function to handle tilt calculation
                const handleTilt = (x, y, rect) => {
                    const centerX = rect.width / 2;
                    const centerY = rect.height / 2;
                    
                    // Calculate rotation angles
                    const rotateX = ((y - centerY) / centerY) * maxTilt;
                    const rotateY = ((x - centerX) / centerX) * -maxTilt; // Negative for correct direction

                    // Apply the 3D transform
                    card.style.transform = `perspective(1000px) rotateX(${rotateX}deg) rotateY(${rotateY}deg) scale(1.02)`;
                };

                // Mouse Events (Desktop)
                card.addEventListener('mousemove', (e) => {
                    const rect = card.getBoundingClientRect();
                    const x = e.clientX - rect.left;
                    const y = e.clientY - rect.top;
                    handleTilt(x, y, rect);
                });

                card.addEventListener('mouseleave', () => {
                    // Reset on mouse leave with transition
                    card.style.transform = 'perspective(1000px) rotateX(0deg) rotateY(0deg) scale(1)';
                });
                
                // Touch Events (Mobile - for a simpler interaction, tilt on press)
                card.addEventListener('touchstart', (e) => {
                    e.preventDefault(); // Prevent accidental scrolling
                    const rect = card.getBoundingClientRect();
                    // Use a fixed tilt angle for touch press/hold
                    const rotateX = 5;
                    const rotateY = -5; 
                    card.style.transform = `perspective(1000px) rotateX(${rotateX}deg) rotateY(${rotateY}deg) scale(1.02)`;
                }, { passive: false });

                card.addEventListener('touchend', () => {
                    // Reset on touch release
                    card.style.transform = 'perspective(1000px) rotateX(0deg) rotateY(0deg) scale(1)';
                });
            });
        }

        // --- 7. Mobile Menu Logic ---
        function setupMobileMenu() {
            const openBtn = document.getElementById('mobile-menu-button');
            const closeBtn = document.getElementById('mobile-menu-close');
            const overlay = document.getElementById('mobile-menu-overlay');
            const links = document.querySelectorAll('.mobile-menu-link');

            function toggleMenu() {
                // Toggle the state of the overlay
                if (overlay.classList.contains('-translate-y-full')) {
                    overlay.classList.remove('-translate-y-full');
                    overlay.classList.add('translate-y-0');
                    document.body.style.overflowY = 'hidden'; // Disable background scroll
                } else {
                    overlay.classList.remove('translate-y-0');
                    overlay.classList.add('-translate-y-full');
                    document.body.style.overflowY = 'auto'; // Enable background scroll
                }
            }

            openBtn.addEventListener('click', toggleMenu);
            closeBtn.addEventListener('click', toggleMenu);
            
            // Close menu when a navigation link is clicked
            links.forEach(link => {
                link.addEventListener('click', toggleMenu);
            });
        }

        // --- Initialize All Functions on Load ---
        window.onload = function() {
            // Set up event listeners for industry tabs
            document.querySelectorAll('.industry-tab').forEach(button => {
                button.addEventListener('click', (e) => {
                    displayIndustry(e.target.getAttribute('data-industry'));
                });
            });
            // Display the first industry by default
            displayIndustry('restaurant');

            // Must run preloader first, which then calls other setup functions
            runPreloader();
            setupParallaxHero();
            setupTiltEffect();
            setupMobileMenu(); // Call new mobile menu logic

            // Only initialize the custom cursor on larger screens
            if (window.innerWidth >= 768) {
                setupCustomCursor();
            }
        };
    </script>
